```

建议集成代码:
    from crypto_wallet.scripts.wallet import HDWallet
    
    # 为每个农场账户派生地址
    wallet = HDWallet("farming-01")
    eth_addr = wallet.derive_address("ethereum", index=0)
    sol_addr = wallet.derive_address("solana", index=0)
```
