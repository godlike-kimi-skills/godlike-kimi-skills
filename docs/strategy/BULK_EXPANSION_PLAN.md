## 快速创建模板 / Fast Creation Template

### 使用脚本批量生成

```python
# tools/bulk-generator/generate-skills.py
# 根据模板快速生成skills

import os
import shutil
from pathlib import Path

def generate_skill(name, category, description_zh, description_en):
    """快速生成一个skill"""
    
    # 复制模板
    template_dir = Path("templates/skill-template")
    skill_dir = Path(f"skills/{name}")
    
    if skill_dir.exists():
        print(f"⚠️ {name} 已存在，跳过")
        return
    
    shutil.copytree(template_dir, skill_dir)
    
    # 修改SKILL.md
    skill_md = skill_dir / "SKILL.md"
    content = skill_md.read_text(encoding='utf-8')
    
    # 替换占位符
    content = content.replace('your-skill-name', name)
    content = content.replace('技能中文名称', description_zh.split('，')[0])
    content = content.replace('Skill English Name', description_en.split(',')[0])
    content = content.replace('所属类别', category)
    
    # 更新description
    content = content.replace(
        '中文描述：这个skill是做什么的',
        f'中文描述：{description_zh}'
    )
    content = content.replace(
        'English description: What this skill does',
        f'English description: {description_en}'
    )
    
    skill_md.write_text(content, encoding='utf-8')
    
    # 更新main.py
    main_py = skill_dir / "scripts/main.py"
    content = main_py.read_text(encoding='utf-8')
    content = content.replace('Skill Core Entry Script', f'{name} Entry Script')
    content = content.replace('中文功能描述', description_zh)
    content = content.replace('English function description', description_en)
    main_py.write_text(content, encoding='utf-8')
    
    print(f"✅ 已生成: {name}")

# 批量生成示例
skills_to_generate = [
    ("file-batch-rename", "file-tools", "批量重命名文件，支持正则表达式和序号", "Batch rename files with regex and numbering"),
    ("file-organizer", "file-tools", "按规则自动整理文件到不同目录", "Automatically organize files into directories by rules"),
    # ... 更多
]

for name, category, desc_zh, desc_en in skills_to_generate:
    generate_skill(name, category, desc_zh, desc_en)
```
